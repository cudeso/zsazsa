{# Sortable table header. Renders a whose link toggles ?sort=&dir=... while preserving any other query args (filters). Reads the current sort from request.args, so the calling route only needs to sort its list accordingly. #} {% macro sort_th(label, key, width='') %} {% set cur = request.args.get('sort', '') %} {% set curdir = request.args.get('dir', 'asc') %} {% set newdir = 'desc' if (cur == key and curdir == 'asc') else 'asc' %} {% set args = dict(request.args.to_dict(), sort=key, dir=newdir) %} {{ label }} {% if cur == key %} {% else %} {% endif %} {% endmacro %}
